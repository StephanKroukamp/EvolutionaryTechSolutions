DROP SCHEMA IF EXISTS TutorBusiness;
CREATE SCHEMA IF NOT EXISTS TutorBusiness;

CREATE TABLE IF NOT EXISTS TutorBusiness.Parent (
  id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  firstName VARCHAR(50) NOT NULL,
  lastName VARCHAR(50) NOT NULL
) ENGINE=INNODB;

CREATE TABLE IF NOT EXISTS TutorBusiness.Student (
  id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
  firstName VARCHAR(50) NOT NULL,
  lastName VARCHAR(50) NOT NULL,
  parentId INT NOT NULL,
  KEY IX_student_parentId (parentId),
  CONSTRAINT FK_student_parent_parentId FOREIGN KEY (parentId) REFERENCES parent (id) ON DELETE RESTRICT
) ENGINE=INNODB;
